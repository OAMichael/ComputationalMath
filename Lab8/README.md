$$
	\begin{cases*}
		y'' - x\sqrt{y} = 0, &emsp;&emsp;&emsp; 0 \leqslant x \leqslant 1, \\

		\displaystyle\int\limits_{0}^{1} y(x)dx = 1.
	\end{cases*}
$$
